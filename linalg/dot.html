import numpy as np
import matplotlib.pyplot as plt

def plane(lim=[-5, 5], xlim=None, ylim=None):
    if lim is not None:
        xlim = ylim = lim
    plt.plot(xlim, [0, 0], color="black")
    plt.plot([0, 0], ylim, color="black")
    plt.xlim(xlim)
    plt.ylim(ylim)
    plt.xticks(np.arange(xlim[0], xlim[1]+1))
    plt.yticks(np.arange(ylim[0], ylim[1]+1))
    plt.grid()
